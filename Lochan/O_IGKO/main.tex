% Define custom environments of Big and Small Roman numbers
\newenvironment{bigroman}[1]
  {
    \begin{enumerate}[label=\Roman*.]
  }
  { \end{enumerate} }

\newenvironment{smallroman}
  {
    \begin{enumerate}[label = (\roman*)]
  }
  { \end{enumerate} }

\newenvironment{bigalphabet}[1]
  {
    \begin{enumerate}[label = (\Alph*), start=#1]
  }
  { \end{enumerate} }

\newenvironment{smallalphabet}[1]
  {
    \begin{enumerate}[label = (\alph*), start=#1]
  }
  { \end{enumerate} }


% Template for Matching MCQ
\newcounter{matchleft}
\newcounter{matchright}

\newenvironment{matchtabularh}{%
  \setcounter{matchleft}{0}%
  \setcounter{matchright}{0}%
  \tabularx{\textwidth}{%
    >{\leavevmode\hbox{} to 1.5em{\stepcounter{matchleft}}}X%
    >{\leavevmode\hbox{} to 1.5em{\stepcounter{matchright}}}X%
    }%
}{\endtabularx}

\newenvironment{matchtabular}{%
  \setcounter{matchleft}{0}%
  \setcounter{matchright}{0}%
  \tabularx{\textwidth}{%
    >{\leavevmode\hbox{} to 1.5em{\stepcounter{matchleft} (\roman{matchleft}) }}X%
    >{\leavevmode\hbox{} to 1.5em{\stepcounter{matchright} (\alph{matchright}) }}X%
    }%
}{\endtabularx}
